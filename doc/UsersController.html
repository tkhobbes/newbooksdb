def create
  @user = User.new(user_params)

  if @user.save
    flash[:success] = 'You successfully signed up'
    redirect_to user_path(@user)
  else
    render :new, status: :unprocessable_entity
  end

end

  else
    render :edit, status: :unprocessable_entity
  end
